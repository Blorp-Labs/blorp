import type { Meta, StoryObj } from "@storybook/react";

import { DetailPostCard, FeedPostCard, getPostProps } from "./post";
import { usePostsStore } from "~/src/stores/posts";
import _ from "lodash";
import { useEffect } from "react";
import * as lemmy from "~/test-utils/lemmy";
import { flattenPost } from "~/src/lib/lemmy/utils";

function LoadPost({ variant }: { variant: "image" | "youtube" | "article" }) {
  const cachePost = usePostsStore((s) => s.cachePost);

  useEffect(() => {
    const postView = lemmy.getPost({
      variant,
    });

    cachePost(
      flattenPost({
        post_view: postView,
      }),
    );
  }, [variant]);

  return null;
}
